package ru.startandroid.aplication_1;

import android.util.Log;
import android.view.MotionEvent;
import android.view.View;
import android.graphics.Canvas;
import android.graphics.Paint;
import android.graphics.Color;
import android.content.Context;

import java.util.Random;

public class Widget1 extends View {
    float density;
    float x;
    float y;
    int block;
    int cell_x;
    int cell_y;
    float ship_x;
    float ship_y;
    float surplus_x;
    float surplus_y;
    boolean destroy;
    boolean milk;
    Random random;
    public Widget1(Context context, float density, int width, int height) {
        super(context);
        this.random = new Random();

        this.density = density;
        this.x = 0;
        this.y = 0;
        this.block = 300;
        this.destroy = false;
        this.milk = false;
        this.cell_x = width / this.block;
        this.cell_y = height / this.block;
        this.surplus_x = width % this.block / 2;
        this.surplus_y = height % this.block;

        this.ship_x = this.random.nextInt(this.cell_x) + this.surplus_x;
        this.ship_y = this.random.nextInt(this.cell_y) + this.surplus_y;
        while (this.ship_y > this.cell_y * this.block) {
            this.ship_x = this.random.nextInt(this.cell_x) + this.surplus_x;
            this.ship_y = this.random.nextInt(this.cell_y) + this.surplus_y;
        }
    }

    @Override
    protected void onDraw(Canvas canvas) {
        super.onDraw(canvas);
        Log.i("ship", "x: " + this.ship_x + ", y: " + this.ship_y);

        Paint paint = new Paint();
        paint.setColor(Color.BLACK);
        paint.setStrokeWidth(1);
        paint.setStyle(Paint.Style.STROKE);
        for (int i = 0; i <= getWidth(); i += this.block) {
            for (int j = 0; j <= getHeight(); j += this.block) {
                canvas.drawLine(this.surplus_x, j + this.surplus_y, (getWidth()  / this.block) * this.block + this.surplus_x, j + this.surplus_y, paint);
                canvas.drawLine(i + this.surplus_x, this.surplus_y, i + this.surplus_x, (getHeight() / this.block) * this.block + this.surplus_y, paint);
            }
        }

        shoot(canvas);

    }
    @Override
    public boolean onTouchEvent(MotionEvent event) {
        if (event.getAction() == MotionEvent.ACTION_DOWN) {
            this.x = ((int) event.getX() / this.block) + this.surplus_x;
            this.y = ((int) event.getY() / this.block) + this.surplus_y;
            Log.i("onTouch", "x: " + this.x + ", y: " + this.y);
            if (this.x - this.surplus_x == this.ship_x - this.surplus_x & this.y - this.surplus_y == this.ship_y - this.surplus_y) {
                this.destroy = true;
                invalidate();
                this.ship_x = this.random.nextInt(this.cell_x) + this.surplus_x;
                this.ship_y = this.random.nextInt(this.cell_y) + this.surplus_y;
                Log.i("while", "shipy: " + this.ship_y + ", y: " + this.cell_y * this.block);

                while (this.ship_y > this.cell_y * this.block) {
                    Log.i("while", "shipy: " + this.ship_y + ", y: " + this.cell_y * this.block);

                    this.ship_x = this.random.nextInt(this.cell_x) + this.surplus_x;
                    this.ship_y = this.random.nextInt(this.cell_y) + this.surplus_y;
                }
            } else {
                this.milk = true;
                this.destroy = false;
                invalidate();
            }
        }
        return false;
    }

    public void shoot(Canvas canvas) {
        if (this.milk) {
            Paint d = new Paint();
            d.setColor(Color.RED);
            d.setStrokeWidth(6);
            d.setStyle(Paint.Style.STROKE);
            Paint paint = new Paint();
            paint.setColor(Color.BLUE);
            paint.setStrokeWidth(4);
            paint.setStyle(Paint.Style.STROKE);
            for (float xx = this.x - 2 - this.surplus_x; xx < this.x + 3 - this.surplus_x; xx++) {
                for (float yy = this.y - 2 - this.surplus_y; yy < this.y + 3 - this.surplus_y; yy++) {
                    if (xx == this.ship_x - this.surplus_x & yy == this.ship_y - this.surplus_y) {
                        canvas.drawCircle(xx * this.block + this.block / 2 + this.surplus_x,yy * this.block + this.block / 2 + this.surplus_y, this.block / 2, d);
                    } else {
                        canvas.drawCircle(xx * this.block + this.block / 2 + this.surplus_x,yy * this.block + this.block / 2 + this.surplus_y, this.block / 4, paint);
                    }
                }
            }
        }
        if (this.destroy) {
            Paint d = new Paint();
            d.setColor(Color.RED);
            d.setStrokeWidth(10);
            d.setStyle(Paint.Style.STROKE);
            canvas.drawLine(this.x * this.block, this.y * this.block, this.x * this.block + this.block, this.y * this.block + this.block, d);
            canvas.drawLine(this.x * this.block, this.y * this.block + this.block, this.x * this.block + this.block, this.y * this.block, d);

        }
    }
}
